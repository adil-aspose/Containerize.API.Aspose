528  if (buffer->get_Length() - offset < count)
529  throw ArgumentException(u"Invalid buffer length");
530 
531  if (count == 0)
532  return 0;
533 
534  if (m_elem_size == 1)
535  {
536  int readed_count = int(m_stream.rdbuf()->sgetn((char_type*)&buffer[offset], count));
537  if (m_wrapping_mode == STDIOStreamWrappingMode::Binary)

540  }
541 
542  Sync();
543  return readed_count;
544  }
545 
546  if (m_wrapping_mode == STDIOStreamWrappingMode::Conversion)
547  {
548  std::vector<char_type> tempbuff(count, 0);
549  int readed_count = int(m_stream.rdbuf()->sgetn(tempbuff.data(), count));
550  auto dst = &buffer[offset];
551  std::for_each(tempbuff.begin(), tempbuff.end(),
552  [&dst, this](char_type& elem) { *(dst++) = uint8_t(elem); });
553 
554  Sync();
555  return readed_count;
556  }
557  else
558  { // m_wrapping_mode == STDIOStreamWrappingMode::Binary
559  uint8_t byte_pos_shift = m_byte_pos % m_elem_size;
560  int new_byte_count = byte_pos_shift + count;
561  bool has_new_byte_pos_shift = new_byte_count % m_elem_size != 0;
562  int elem_count = new_byte_count / m_elem_size + int(has_new_byte_pos_shift);
563 
564  std::vector<char_type> tempbuff(elem_count, 0);
565  int readed_elem_count = int(m_stream.rdbuf()->sgetn(tempbuff.data(), elem_count));
566  int readed_byte_count;
567  if (readed_elem_count != elem_count)
568  {
569  readed_byte_count = readed_elem_count * m_elem_size - byte_pos_shift;
570  }
571  else
572  {
573  readed_byte_count = count;
574  if (has_new_byte_pos_shift)
575  {
576  m_stream.rdbuf()->pubseekoff(-1, std::ios_base::cur, std::ios_base::in);
577  }
578  }
579 
580  m_byte_pos += readed_byte_count;
581 
582  const char_type* source = &tempbuff[byte_pos_shift];
583  uint8_t* destination = &buffer[offset];

586  DecodeElem(traits_type::to_int_type(*source));

832  }
833  }

838  std::vector<char_type> tempbuff(elem_count, 0);
839  char_type* destination = tempbuff.data();

843  }

846  }

851  { // There are unwritten bytes. Write bytes in decoded_elem, encode it and put result in the stream.
852  if (m_decoded_elem == nullptr)

855  }
856

858  {
859  m_decoded_elem[byte_pos_shift] = *source;
860  }
861

863  m_stream.rdbuf()->pubseekoff(-1, std::ios_base::cur, std::ios_base::out);
864  }
