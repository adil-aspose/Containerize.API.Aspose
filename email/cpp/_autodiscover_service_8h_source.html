698  template <typename TSettings>
699  TSettings InternalGetLegacyUserSettings(System::String emailAddress, System::SharedPtr<System::Collections::Generic::ListExt<System::String>> redirectionEmailAddresses, int32_t& currentHop)
700  {
701  typedef ConfigurationSettingsBase BaseT_ConfigurationSettingsBase;
702  assert_is_base_of(BaseT_ConfigurationSettingsBase, TSettings);
703  assert_is_constructable(TSettings);
704 
705  System::String domainName = DomainFromEmailAddress(emailAddress);
706 
707  int32_t scpUrlCount;
708  System::SharedPtr<System::Collections::Generic::ListExt<System::SharedPtr<System::Uri>>> urls = this->GetAutodiscoverServiceUrls(domainName, scpUrlCount);
709 
710  if (urls->get_Count() == 0)
711  {
712  throw AutodiscoverException(u"SR.AutodiscoverServiceRequestRequiresDomainOrUrl");
713  }
714 
715  // Assume caller is not inside the Intranet, regardless of whether SCP Urls
716  // were returned or not. SCP Urls are only relevant if one of them returns
717  // valid Autodiscover settings.
718  this->isExternal = true;
719 
720  int32_t currentUrlIndex = 0;
721 
722  // Used to save exception for later reporting.
723  System::Exception delayedException = nullptr;
724  TSettings settings;
725 
726  do
727  {
728  System::SharedPtr<System::Uri> autodiscoverUrl = urls->idx_get(currentUrlIndex);
729  bool isScpUrl = currentUrlIndex < scpUrlCount;
730 
731  try
732  {
733  settings = this->template GetLegacyUserSettingsAtUrl<TSettings>(emailAddress, autodiscoverUrl);
734 
735  switch (settings->get_ResponseType())
736  {
737  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::Success:
738  if (isScpUrl)
739  {
740  this->set_IsExternal(false);
741  }
742  this->set_Url(autodiscoverUrl);
743  return settings;
744 
745  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::RedirectUrl:
746  if (currentHop < AutodiscoverMaxRedirections)
747  {
748  currentHop++;
749  Log(System::String::Format(u"Autodiscover service returned redirection URL '{0}'.", settings->get_RedirectTarget()));
750 
751  urls->idx_set(currentUrlIndex, System::MakeObject<System::Uri>(settings->get_RedirectTarget()));
752  break;
753  }
754  else
755  {
756  throw AutodiscoverException(u"SR.MaximumRedirectionHopsExceeded");
757  }
758 
759  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::RedirectAddress:
760  if (currentHop < AutodiscoverMaxRedirections)
761  {
762  currentHop++;
763  Log(System::String::Format(u"Autodiscover service returned redirection email address '{0}'.", settings->get_RedirectTarget()));
764 
765  // If this email address was already tried, we may have a loop
766  // in SCP lookups. Disable consideration of SCP records.
767  this->DisableScpLookupIfDuplicateRedirection(settings->get_RedirectTarget(), redirectionEmailAddresses);
768 
769  return this->template InternalGetLegacyUserSettings<TSettings>(settings->get_RedirectTarget(), redirectionEmailAddresses, System::Ref(currentHop));
770  }
771  else
772  {
773  throw AutodiscoverException(u"SR.MaximumRedirectionHopsExceeded");
774  }
775 
776  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::Error:
777  if (isScpUrl)
778  {
779  Log(u"Error returned by Autodiscover service found via SCP, treating as inconclusive.");
780 
781  delayedException = AutodiscoverException(u"SR.AutodiscoverError", settings->get_Error());
782  currentUrlIndex++;
783  }
784  else
785  {
786  throw AutodiscoverException(u"SR.AutodiscoverError", settings->get_Error());
787  }
788  break;
789 
790  default:
791  Assert(false, u"Autodiscover.GetConfigurationSettings", u"An unexpected error has occurred. This code path should never be reached.");
792  break;
793 
794  }
795  }
796  catch (System::Net::WebException& ex)
797  {
798  if (ex->get_Response() != nullptr)
799  {
800  System::SharedPtr<System::Net::HttpWebResponse> response = CreateExceptionResponse(ex);
801  System::SharedPtr<System::Uri> redirectUrl;
802  if (this->TryGetRedirectionResponse(response, redirectUrl))
803  {
804  Log(System::String::Format(u"Host returned a redirection to url {0}", redirectUrl));
805 
806  currentHop++;
807  urls->idx_set(currentUrlIndex, redirectUrl);
808  }
809  else
810  {
811  this->ProcessHttpErrorResponse(response, ex);
812 
813  Log(System::String::Format(u"{0} failed: {1} ({2})", url, System::ObjectExt::GetType(ex).get_Name(), ex->get_Message()));
814 
815  // The url did not work, let's try the next.
816  currentUrlIndex++;
817  }
818  }
819  else
820  {
821  Log(System::String::Format(u"{0} failed: {1} ({2})", url, System::ObjectExt::GetType(ex).get_Name(), ex->get_Message()));
822 
823  // The url did not work, let's try the next.
824  currentUrlIndex++;
825  }
826  }
827  catch (System::Xml::XmlException& ex)
828  {
829  Log(System::String::Format(u"{0} failed: XML parsing error: {1}", url, ex->get_Message()));
830 
831  // The content at the URL wasn't a valid response, let's try the next.
832  currentUrlIndex++;
833  }
834  catch (System::IO::IOException& ex)
835  {
836  Log(System::String::Format(u"{0} failed: I/O error: {1}", url, ex->get_Message()));
837 
838  // The content at the URL wasn't a valid response, let's try the next.
839  currentUrlIndex++;
840  }
841 
842  } while (currentUrlIndex < urls->get_Count());
843 
844  // If we got this far it's because none of the URLs we tried have worked. As a next-to-last chance, use GetRedirectUrl to
845  // try to get a redirection URL using an HTTP GET on a non-SSL Autodiscover endpoint. If successful, use this
846  // redirection URL to get the configuration settings for this email address. (This will be a common scenario for
847  // DataCenter deployments).
848  System::SharedPtr<System::Uri> redirectionUrl = this->GetRedirectUrl(domainName);
849  if (redirectionUrl != nullptr && this->template TryLastChanceHostRedirection<TSettings>(emailAddress, redirectionUrl, settings))
850  {
851  return settings;
852  }
853  else
854  {
855  // Getting a redirection URL from an HTTP GET failed too. As a last chance, try to get an appropriate SRV Record
856  // using DnsQuery. If successful, use this redirection URL to get the configuration settings for this email address.
857  redirectionUrl = this->GetRedirectionUrlFromDnsSrvRecord(domainName);
858  if (redirectionUrl != nullptr && this->template TryLastChanceHostRedirection<TSettings>(emailAddress, redirectionUrl, settings))
859  {
860  return settings;
861  }
862  else if (delayedException != nullptr)
863  {
864  throw AutodiscoverException(delayedException->get_Message(), delayedException);
865  }
866  else
867  {
868  throw AutodiscoverException(u"SR.AutodiscoverCouldNotBeLocated");
869  }
870  }

881  bool TryLastChanceHostRedirection(System::String emailAddress, System::SharedPtr<System::Uri> redirectionUrl, TSettings& settings)
882  {
883  typedef ConfigurationSettingsBase BaseT_ConfigurationSettingsBase;
884  assert_is_base_of(BaseT_ConfigurationSettingsBase, TSettings);
885  assert_is_constructable(TSettings);
886 
887  settings.reset();
888 
889  System::SharedPtr<System::Collections::Generic::ListExt<System::String>> redirectionEmailAddresses = System::MakeObject<System::Collections::Generic::ListExt<System::String>>();
890 
891  // Bug 60274: Performing a non-SSL HTTP GET to retrieve a redirection URL is potentially unsafe. We allow the caller
892  // to specify delegate to be called to determine whether we are allowed to use the redirection URL.
893  if (this->CallRedirectionUrlValidationCallback(System::ObjectExt::ToString(redirectionUrl)))
894  {
895  for (int32_t currentHop = 0; currentHop < AutodiscoverService::AutodiscoverMaxRedirections; currentHop++)
896  {
897  try
898  {
899  settings = this->template GetLegacyUserSettingsAtUrl<TSettings>(emailAddress, redirectionUrl);
900 
901  switch (settings->get_ResponseType())
902  {
903  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::Success:
904  return true;
905 
906  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::Error:
907  throw AutodiscoverException(u"SR.AutodiscoverError", settings->get_Error());
908 
909  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::RedirectAddress:
910  this->DisableScpLookupIfDuplicateRedirection(settings->get_RedirectTarget(), redirectionEmailAddresses);
911  settings = this->template InternalGetLegacyUserSettings<TSettings>(settings->get_RedirectTarget(), redirectionEmailAddresses, System::Ref(currentHop));
912  return true;
913 
914  case Aspose::Email::Clients::Exchange::AutodiscoverResponseType::RedirectUrl:
915  try
916  {
917  redirectionUrl = System::MakeObject<System::Uri>(settings->get_RedirectTarget());
918  }
919  catch (System::UriFormatException& )
920  {
921  Log(System::String::Format(u"Service returned invalid redirection URL {0}", settings->get_RedirectTarget()));
922  return false;
923  }
924 
925  break;
926 
927  default:
928  {
929  System::String failureMessage = System::String::Format(u"Autodiscover call at {0} failed with error {1}, target {2}", redirectionUrl, settings->get_ResponseType(), settings->get_RedirectTarget());
930  Log(failureMessage);
931  return false;
932  }
933 
934  }
935  }
936  catch (System::Net::WebException& ex)
937  {
938  if (ex->get_Response() != nullptr)
939  {
940  System::SharedPtr<System::Net::HttpWebResponse> response = CreateExceptionResponse(ex);
941  if (this->TryGetRedirectionResponse(response, redirectionUrl))
942  {
943  Log(System::String::Format(u"Host returned a redirection to url {0}", redirectionUrl));
944  continue;
945  }
946  else
947  {
948  this->ProcessHttpErrorResponse(response, ex);
949  }
950  }
951 
952  Log(System::String::Format(u"{0} failed: {1} ({2})", url, System::ObjectExt::GetType(ex).get_Name(), ex->get_Message()));
953 
954  return false;
955  }
956  catch (System::Xml::XmlException& ex)
957  {
958  // If the response is malformed, it wasn't a valid Autodiscover endpoint.
959  Log(System::String::Format(u"{0} failed: XML parsing error: {1}", redirectionUrl, ex->get_Message()));
960  return false;
961  }
962  catch (System::IO::IOException& ex)
963  {
964  Log(System::String::Format(u"{0} failed: I/O error: {1}", redirectionUrl, ex->get_Message()));
965  return false;
966  }
967 
968  }
969  }
970 
971  return false;
